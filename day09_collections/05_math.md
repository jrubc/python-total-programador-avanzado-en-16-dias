# Math

```python
import math

resultado = math.floor(89.665)
print(resultado)

resultado = math.ceil(89.665)
print(resultado)

resultado = math.pi
print(resultado)

resultado = math.log(100, 2)
print(resultado)

resultado = math.tan(250)
print(resultado)

resultado = math.cos(250)
print(resultado)
```
